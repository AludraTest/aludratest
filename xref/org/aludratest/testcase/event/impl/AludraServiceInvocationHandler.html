16  package org.aludratest.testcase.event.impl;
17  
18  import java.lang.reflect.InvocationHandler;
19  import java.lang.reflect.Method;
20  import java.util.HashMap;
21  import java.util.Map;
22  
23  import org.aludratest.service.Action;
24  import org.aludratest.service.AludraService;
25  import org.aludratest.service.ComponentId;
26  import org.aludratest.service.SystemConnector;
27  import org.aludratest.service.util.SystemConnectorProxy;
28  import org.aludratest.testcase.AludraTestContext;
29  import org.aludratest.util.AludraTestUtil;
30  import org.aludratest.util.ControlFlowHandler;
31  
32  /** {@link InvocationHandler} implementation for the dynamic proxy API. It is intended to wrap {@link AludraService} objects X and
33   * add features in an aspect-like way to them: Any object Y returned by from a call to X' methods perform(), verify() or check()
34   * is wrapped with a {@link ControlFlowHandler} proxy that controls execution.
35   * @author Volker Bergmann */
36  public class AludraServiceInvocationHandler implements InvocationHandler {
37  
38      /** The service object to wrap. */
39      private Object realObject;
40  
41      /** A system connector to provide to the {@link ControlFlowHandler} for checking asynchronous error occurrences. Usage of an
42       * {@link SystemConnectorProxy} allows me to pass the connector proxy to the service proxies and exchange the real connector
43       * afterwards. */
44      private SystemConnectorProxy systemConnector;
45  
46      /** The {@link ComponentId} of the instance. */
47      private ComponentId<? extends AludraService> serviceId;
48  
49      /** A pool of previously used proxies for reuse. */
50      private Map<String, Object> proxyPool;
51  
52      private AludraTestContext context;
53  
54      /** Constructor
55       * @param realObject the service object to wrap
56       * @param serviceId
57       * @param context The AludraTest context to use to e.g. retrieve configuration. */
58      public AludraServiceInvocationHandler(Object realObject, ComponentId<? extends AludraService> serviceId,
59              AludraTestContext context) {
60          this.realObject = realObject;
61          this.systemConnector = new SystemConnectorProxy(null);
62          this.serviceId = serviceId;
63          this.context = context;
64          this.proxyPool = new HashMap<String, Object>();
65      }
66  
67      /**
68       * Performs the actual method invocation on the real object,
69       * and on calls to methods perform(), verify() or check() it
70       * creates or reuses logging proxies that track the
71       * invocation information with log4testing.
72       */
73      @Override
74      public Object invoke(Object proxy, Method method, Object[] args) throws Throwable { //NOSONAR
75          // Note: The following code provokes 'Security - Array is stored directly' in SonarCube,
76          //       but the behavior is correct, since this is wrapper code which has to forward
77          //       invocation data unchanged.
78          String methodName = method.getName();
79          if ("setSystemConnector".equals(methodName)) {
80              SystemConnector newConnector = (SystemConnector) args[0];
81              this.systemConnector.setRealSystemConnector(newConnector);
82              return method.invoke(realObject, this.systemConnector);
83          } else if (methodName.equals("perform") || methodName.equals("verify") || methodName.equals("check")) {
84              Object invocationResult = method.invoke(realObject, args);
85              Class<?> interfaceType = method.getReturnType();
86              Object wrappedResult = getOrCreateProxy(invocationResult, interfaceType);
87              if (wrappedResult instanceof Action) {
88                  ((Action) wrappedResult).setSystemConnector(this.systemConnector);
89              }
90              return wrappedResult;
91          } else {
92              return method.invoke(realObject, args);
93          }
94      }
95  
96  
97      // private helper methods --------------------------------------------------
98  
99      private Object getOrCreateProxy(Object realResult, Class<?> interfaceType) {
100         String objectId = interfaceType.getName() + "@" + System.identityHashCode(realResult);
101         Object result = proxyPool.get(objectId);
102         if (result == null) {
103             result = createProxy(realResult, interfaceType, objectId);
104         }
105         return result;
106     }
107 
108     @SuppressWarnings({ "unchecked", "rawtypes" })
109     private Object createProxy(Object realObject, Class interfaceType, String objectId) {
110         Object result = AludraTestUtil.wrapWithControlFlowHandler(realObject, interfaceType, serviceId, systemConnector, context);
111         proxyPool.put(objectId, result);
112         return result;
113     }
114 
115 }
