16  package org.aludratest.service.file.impl;
17  
18  import org.aludratest.config.Preferences;
19  import org.aludratest.config.ValidatingPreferencesWrapper;
20  import org.aludratest.exception.AutomationException;
21  import org.aludratest.service.AludraCloseable;
22  import org.aludratest.service.file.FileService;
23  import org.apache.commons.vfs2.FileName;
24  import org.apache.commons.vfs2.FileObject;
25  import org.apache.commons.vfs2.FileSystemException;
26  import org.apache.commons.vfs2.FileSystemManager;
27  import org.apache.commons.vfs2.FileSystemOptions;
28  import org.apache.commons.vfs2.UserAuthenticator;
29  import org.apache.commons.vfs2.auth.StaticUserAuthenticator;
30  import org.apache.commons.vfs2.impl.DefaultFileSystemConfigBuilder;
31  import org.apache.commons.vfs2.impl.StandardFileSystemManager;
32  import org.apache.commons.vfs2.provider.ftp.FtpFileSystemConfigBuilder;
33  
34  /**
35   * Provides configuration for the {@link FileService}.
36   * @author Volker Bergmann
37   */
38  public class FileServiceConfiguration implements AludraCloseable {
39  
40      static final int DEFAULT_WAIT_MAX_RETRIES = 15;
41  
42      static final int DEFAULT_WAIT_TIMEOUT = 30000;
43  
44      private ValidatingPreferencesWrapper configuration;
45  
46      /** Common-vfs' {@link FileSystemManager}. */
47      private StandardFileSystemManager manager;
48  
49      /** The root folder used by this service instance. */
50      private FileObject rootFolder;
51  
52      /** Creates a new FileServiceConfiguration object which wraps the given Preferences object.
53       * 
54       * @param configuration Preferences configuration object to wrap.
55       * 
56       * @throws FileSystemException If an exception occurs when applying the configuration to the VFS Config Builders. */
57      public FileServiceConfiguration(Preferences configuration) throws FileSystemException {
58          this.configuration = new ValidatingPreferencesWrapper(configuration);
59  
60          // Configure secured access
61          FileSystemOptions fileSystemOptions = new FileSystemOptions();
62          String protocol = getProtocol();
63          String baseUrl = getBaseUrl();
64          String user = getUser();
65          String password = getPassword();
66  
67          if (user != null || password != null) {
68              UserAuthenticator authenticator = new StaticUserAuthenticator(protocol, user, password);
69              DefaultFileSystemConfigBuilder builder = DefaultFileSystemConfigBuilder.getInstance();
70              builder.setUserAuthenticator(fileSystemOptions, authenticator);
71          }
72          if ("ftp".equals(protocol)) {
73              FtpFileSystemConfigBuilder builder = FtpFileSystemConfigBuilder.getInstance();
74              builder.setUserDirIsRoot(fileSystemOptions, false);
75              builder.setDataTimeout(fileSystemOptions, getTimeout());
76              builder.setSoTimeout(fileSystemOptions, getTimeout());
77              builder.setPassiveMode(fileSystemOptions, true);
78          }
79  
80          // configure FileObject for root folder
81          this.manager = new StandardFileSystemManager();
82          this.manager.init();
83          this.rootFolder = manager.resolveFile(protocol + "://" + baseUrl, fileSystemOptions);
84  
85          // access all configuration element in order to verify a complete configuration
86          getEncoding();
87          getLinefeed();
88          getPollingDelay();
89          getWaitMaxRetries();
90          getTimeout();
91          getRootFolder();
92          getFileObject("/");
93          getHost();
94          isWritingPermitted();
95  
96      }
97  
98      /** @return the name of the used protocol: file, ftp, sftp, http or https. */
99      public final String getProtocol() {
100         return configuration.getRequiredStringValue("protocol");
101     }
102 
103     /** @return the base URL of the service. */
104     public final String getBaseUrl() {
105         return configuration.getRequiredStringValue("base.url").replace('\\', '/');
106     }
107 
108     /** @return the file encoding used on the file system. */
109     public final String getEncoding() {
110         return configuration.getRequiredStringValue("encoding");
111     }
112 
113     /** @return the linefeed character(s) used on the file system. */
114     public final String getLinefeed() {
115         return Linefeed.valueOf(configuration.getRequiredStringValue("linefeed")).getChars();
116     }
117 
118     /** @return the user name for login. */
119     public final String getUser() {
120         return configuration.getStringValue("user");
121     }
122 
123     /** @return the password for login. */
124     public final String getPassword() {
125         return configuration.getStringValue("password");
126     }
127 
128     /** @return the delay to use between polls. */
129     public final int getPollingDelay() {
130         return getTimeout() / getWaitMaxRetries();
131     }
132 
133     /** @return the maximum number of polls to try. */
134     private final int getWaitMaxRetries() {
135         return configuration.getIntValue("wait.max.retries", DEFAULT_WAIT_MAX_RETRIES);
136     }
137 
138     /** @return the maximum time to wait when polling. */
139     public final int getTimeout() {
140         return configuration.getIntValue("wait.timeout", DEFAULT_WAIT_TIMEOUT);
141     }
142 
143     /** @return the root folder of the service. */
144     public final FileObject getRootFolder() {
145         return rootFolder;
146     }
147 
148     /** @return true if writing is permitted on the file system, otherwise false */
149     public final boolean isWritingPermitted() {
150         return configuration.getBooleanValue("writing.permitted", false);
151     }
152 
153     /** @return the host */
154     public final String getHost() {
155         String baseUrl = getBaseUrl();
156         int sep = baseUrl.indexOf('/');
157         String host = sep > 0 ? baseUrl.substring(0, sep) : "localhost";
158         return host;
159     }
160 
161     /** @return a commons-vfs {@link FileObject} for the given path.
162      *  @param filePath the path of the requested file
163      */
164     public final FileObject getFileObject(String filePath) {
165         try {
166             String path = (filePath.charAt(0) == '/' ? (filePath.length() == 1 ? "." : filePath.substring(1)) : filePath);
167             return manager.resolveFile(rootFolder, path);
168         } catch (FileSystemException e) {
169             throw new AutomationException("Error accessing file", e);
170         }
171     }
172 
173     /**
174      * @param file the file
175      * @return the full path
176      */
177     public final String pathFromRoot(FileObject file) {
178         FileName rootFolderName = rootFolder.getName();
179         FileName fileName = file.getName();
180         try {
181             return rootFolderName.getRelativeName(fileName);
182         } catch (FileSystemException e) {
183             throw new AutomationException("Error resolving relative path " +
184                     rootFolderName.getPath() + " -> " + fileName.getPath(), e);
185         }
186     }
187 
188     @Override
189     public void close() {
190         this.manager.close();
191     }
192 
193 }
