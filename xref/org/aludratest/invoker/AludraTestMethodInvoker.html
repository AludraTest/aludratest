16  package org.aludratest.invoker;
17  
18  import java.lang.reflect.Method;
19  import java.lang.reflect.Modifier;
20  
21  import org.aludratest.exception.AutomationException;
22  import org.aludratest.exception.TechnicalException;
23  import org.aludratest.testcase.After;
24  import org.aludratest.testcase.AludraTestCase;
25  import org.aludratest.testcase.AludraTestContext;
26  import org.aludratest.testcase.Before;
27  import org.databene.commons.ArrayFormat;
28  import org.databene.commons.Assert;
29  
30  /** Default {@link TestInvoker} implementation for AludraTest. First it calls the {@literal @}Before methods, then the test method
31   * itself, then the {@literal @}After methods, similar to the behavior familiar from JUnit.
32   * @author Volker Bergmann */
33  public class AludraTestMethodInvoker implements TestInvoker {
34  
35      /** An instance of the test class to invoke. */
36      private AludraTestCase testObject;
37  
38      /** The method to invoke in the test class instance. */
39      private Method method;
40  
41      /** The arguments to provide to the test method. */
42      private Object[] args;
43  
44      /** Constructor receiving values for all attributes of same name.
45       * @param testObject the object on which to invoke the test method
46       * @param method the test method to invoke
47       * @param args the arguments to pass on method invocation */
48      public AludraTestMethodInvoker(AludraTestCase testObject, Method method, Object[] args) {
49          // check preconditions
50          Assert.notNull(testObject, "testObject");
51          Assert.notNull(method, "method");
52          int expectedArgCount = method.getParameterTypes().length;
53          int providedArgCount = (args != null ? args.length : 0);
54          if (expectedArgCount != providedArgCount) {
55              throw new IllegalArgumentException("Method" + method.getName() + " expects " + expectedArgCount + " parameters, " + "but got " + providedArgCount);
56          }
57  
58          // assign fields
59          this.testObject = testObject;
60          this.method = method;
61          this.args = (args != null ? args.clone() : null);
62      }
63  
64      /** Sets the testCase property on the {@link #testObject}. */
65      @Override
66      public void setContext(AludraTestContext context) {
67          this.testObject.setContext(context);
68      }
69  
70      /** First executes all {@literal @}Before methods,
71       *  then the test method and finally all {@literal @}After
72       *  methods on the {@link #testObject}  */
73      @Override
74      public void invoke() throws Exception { //NOSONAR
75          executeBefores(testObject.getClass());
76          try {
77              this.method.invoke(testObject, args);
78          } finally {
79              executeAfters(testObject.getClass());
80          }
81      }
82  
83      @Override
84      public Class<?> getTestClass() {
85          return testObject.getClass();
86      }
87  
88      /** Creates a String representation of the object. */
89      @Override
90      public String toString() {
91          return method.getDeclaringClass().getName() + '.' + method.getName() + '(' + ArrayFormat.format(args) + ')';
92      }
93  
94      // private helper methods ------------------------------------------------------------------------------------------
95  
96      /** Executes all {@literal @}Before methods of the {@link #testObject}
97       *  and its parent classes (parent classes first). */
98      private void executeBefores(Class<?> type) {
99          Class<?> superclass = type.getSuperclass();
100         if (!Object.class.equals(superclass)) {
101             executeBefores(superclass);
102         }
103         for (Method method : type.getDeclaredMethods()) {
104             Before before = method.getAnnotation(Before.class);
105             if (before != null) {
106                 verifyBeforeAfterMethod(method);
107                 invokeObjectMethod(method, testObject);
108             }
109         }
110     }
111 
112     /** Executes all {@literal @}After methods of the {@link #testObject}
113      *  and its parent classes (child classes first). */
114     private void executeAfters(Class<?> type) {
115         for (Method method : type.getDeclaredMethods()) {
116             After after = method.getAnnotation(After.class);
117             if (after != null) {
118                 verifyBeforeAfterMethod(method);
119                 invokeObjectMethod(method, testObject);
120             }
121         }
122         Class<?> superclass = type.getSuperclass();
123         if (!Object.class.equals(superclass)) {
124             executeAfters(superclass);
125         }
126     }
127 
128     /** Invokes the given {@literal @}Before or {@literal @}After method
129      * on the given object */
130     private static void invokeObjectMethod(Method method, Object targetObject) {
131         try {
132             method.invoke(targetObject);
133         } catch (Exception e) {
134             throw new TechnicalException("Before or After method fails! " + "Please contact Testautomation Team! \n" + "Message: " + e.getMessage() + "\n" + "Cause: " + e.getCause().getMessage(), e);
135         }
136     }
137 
138     /** Verifies that a {@literal @}Before or {@literal @}After method
139      *  fulfills all syntactical requirements to be invoked properly. */
140     private static void verifyBeforeAfterMethod(Method method) {
141         if ((method.getModifiers() & Modifier.PUBLIC) != Modifier.PUBLIC) {
142             throw new AutomationException("Before/After methods must be public: " + method);
143         }
144         if ((method.getModifiers() & Modifier.STATIC) == Modifier.STATIC) {
145             throw new AutomationException("Before/After methods must not be static: " + method);
146         }
147         Class<?>[] parameterTypes = method.getParameterTypes();
148         if (parameterTypes != null && parameterTypes.length > 0) {
149             throw new AutomationException("Before/After methods must not have parameters: " + method);
150         }
151     }
152 
153 }
