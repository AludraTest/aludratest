16  package org.aludratest.util;
17  
18  import org.apache.poi.ss.usermodel.Cell;
19  import org.apache.poi.ss.usermodel.Row;
20  import org.apache.poi.ss.usermodel.Sheet;
21  
22  /** Provides Excel-related utility methods.
23   * @author Volker Bergmann */
24  public class ExcelUtil {
25  
26      /** Searches a {@link Row} for a {@link Cell} with the provided text and returns its index.
27       * @param text the text to search
28       * @param row the row in which to search
29       * @return the index of the located cell or -1 if the text was not found in the row */
30      public static int findCellWithText(String text, Row row) {
31          int lastCellIndex = row.getLastCellNum();
32          for (int i = 0; i < lastCellIndex; i++) {
33              Cell cell = row.getCell(i);
34              if (cell != null && cell.getCellType() == Cell.CELL_TYPE_STRING && text.equals(cell.getStringCellValue())) {
35                  return i;
36              }
37          }
38          return -1;
39      }
40  
41      /** Inserts a column without cells into a sheet at the given index.
42       * @param sheet the sheet in which to insert the column
43       * @param insertionIndex the column index at which to insert the new column */
44      public static void insertEmptyColumn(Sheet sheet, int insertionIndex) {
45          for (int i = sheet.getLastRowNum(); i >= 0; i--) {
46              Row row = sheet.getRow(i);
47              short lastCellNum = row.getLastCellNum();
48              if (lastCellNum >= insertionIndex) {
49                  shiftCellsRight(row, insertionIndex, lastCellNum);
50              }
51          }
52      }
53  
54      // private helpers ---------------------------------------------------------
55  
56      private static void shiftCellsRight(Row row, int firstCellNum, short lastCellNum) {
57          for (int i = lastCellNum; i >= firstCellNum; i--) {
58              moveCell(row, i, i + 1);
59          }
60      }
61  
62      private static void moveCell(Row row, int fromIndex, int toIndex) {
63          Cell oldCell = row.getCell(fromIndex);
64          if (oldCell != null) {
65              Cell newCell = row.createCell(toIndex, oldCell.getCellType());
66              newCell.setCellStyle(oldCell.getCellStyle());
67  
68              if (oldCell.getCellType() == Cell.CELL_TYPE_NUMERIC) {
69                  newCell.setCellValue(oldCell.getNumericCellValue());
70              }
71              else if (oldCell.getCellType() == Cell.CELL_TYPE_BOOLEAN) {
72                  newCell.setCellValue(oldCell.getBooleanCellValue());
73              }
74              else if (oldCell.getCellType() == Cell.CELL_TYPE_FORMULA) {
75                  newCell.setCellFormula(oldCell.getCellFormula());
76              }
77              else if (oldCell.getCellType() == Cell.CELL_TYPE_STRING) {
78                  // throws exception for other types
79                  newCell.setCellValue(oldCell.getStringCellValue());
80              }
81              row.removeCell(oldCell);
82          }
83      }
84  
85  }
