38  /**
39   * Tests the {@link FileService} with the SFTP protocol.
40   * @author Volker Bergmann
41   */
42  @SuppressWarnings("javadoc")
43  public class SftpFileServiceTest extends AbstractLocalFileServiceTest {
44  
45      private SshServer sshd;
46  
47      @Override
48      public void prepareFiles() throws Exception {
49          super.prepareFiles();
50          setupSftpServer();
51      }
52  
53      public void setupSftpServer() throws Exception {
54          sshd = SshServer.setUpDefaultServer();
55          sshd.setHost("127.0.0.1");
56          sshd.setPort(4922);
57          sshd.setKeyPairProvider(new SimpleGeneratorHostKeyProvider("target/hostkey.ser"));
58          sshd.setPasswordAuthenticator(new PasswordAuthenticator() {
59              @Override
60              public boolean authenticate(String username, String password, ServerSession session) {
61                  return "ftpuser".equals(username) && "topsecret".equals(password);
62              }
63          });
64  
65          List<NamedFactory<UserAuth>> userAuthFactories = new ArrayList<NamedFactory<UserAuth>>();
66          userAuthFactories.add(new UserAuthPassword.Factory());
67          sshd.setUserAuthFactories(userAuthFactories);
68  
69          sshd.setCommandFactory(new ScpCommandFactory());
70  
71          List<NamedFactory<Command>> namedFactoryList = new ArrayList<NamedFactory<Command>>();
72          namedFactoryList.add(new SftpSubsystem.Factory());
73          sshd.setSubsystemFactories(namedFactoryList);
74  
75          // prepare directory for test files
76          VirtualFileSystemFactory fileSystemFactory = new VirtualFileSystemFactory(ROOT.getAbsolutePath());
77          sshd.setFileSystemFactory(fileSystemFactory);
78  
79          sshd.start();
80      }
81  
82      @Override
83      public void closeTestCase() {
84          try {
85              if (sshd != null)
86                  sshd.stop();
87          }
88          catch (Exception e) {
89              Assert.fail("Unexpected exception when closing SFTP server: " + e.getMessage());
90          }
91          super.closeTestCase();
92      }
93  
94      @Test
95      public void testGetRootChildren() {
96          FileService service = getService(FileService.class, "sftptest");
97          List<String> list = service.perform().getChildren("/");
98          assertTrue(list.size() > 0);
99      }
100 
101     @Test
102     public void testGetSubfolderChildren() {
103         FileService service = getService(FileService.class, "sftptest");
104         List<String> list = service.perform().getChildren("sub");
105         assertTrue(list.size() == 1);
106         assertEquals("sub/subfile.txt", list.get(0));
107     }
108 
109     @Test
110     public void testgetInputStream() {
111         FileService service = getService(FileService.class, "sftptest");
112         service.perform().readTextFile("file.txt");
113     }
114 
115     @Test
116     public void testWrite() {
117         FileService service = getService(FileService.class, "sftptest");
118         service.perform().writeTextFile("aludratest.txt", "This is a test for AludraTest's FileService", true);
119     }

121 }
