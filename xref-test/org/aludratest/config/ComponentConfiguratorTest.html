16  package org.aludratest.config;
17  
18  import static org.junit.Assert.assertEquals;
19  import static org.junit.Assert.assertFalse;
20  import static org.junit.Assert.assertTrue;
21  
22  import java.util.Arrays;
23  
24  import org.aludratest.AludraTest;
25  import org.junit.Test;
26  import org.test.testclasses.service.MoreComplexComponent;
27  import org.test.testclasses.service.MoreComplexComponentImpl;
28  
29  public class ComponentConfiguratorTest {
30  
31      @Test
32      public void testAnnotationLoad() {
33          String propName = "ALUDRATEST_CONFIG/aludraservice/" + MoreComplexComponent.class.getName();
34          System.setProperty(propName, MoreComplexComponentImpl.class.getName());
35  
36          // init AludraTest framework
37          AludraTest aludraTest = AludraTest.startFramework();
38  
39          // retrieve an MCC
40          MoreComplexComponent comp = aludraTest.getServiceManager().newImplementorInstance(MoreComplexComponent.class);
41          assertTrue(comp instanceof MoreComplexComponentImpl);
42          MoreComplexComponentImpl confComp = (MoreComplexComponentImpl) comp;
43          assertTrue(confComp.configured);
44  
45          // assertions against Preferences object
46          MutablePreferences prefs = confComp.getDefaultPreferences();
47  
48          assertEquals(1, prefs.getIntValue("testvalue1"));
49          assertEquals(true, prefs.getBooleanValue("testvalue2"));
50          assertFalse(Arrays.asList(prefs.getKeyNames()).contains("testvalue3"));
51          assertEquals("TEST", prefs.getStringValue("atestvalue"));
52          assertEquals(2.0, prefs.getFloatValue("implProp"), 0.01);
53          assertEquals("!", prefs.getStringValue("dynamicValue"));
54  
55          System.getProperties().remove(propName);
56          aludraTest.stopFramework();
57      }
58  
59  }
